# === One-step install (run from this repo root) ===
#   cd verl-tool
#
#   # MUST unset BEFORE starting Ray — Ray workers inherit and crash with:
#   # "ValueError: Please don't set ROCR_VISIBLE_DEVICES when HIP/CUDA_VISIBLE_DEVICES is set."
#   unset ROCR_VISIBLE_DEVICES
#
#   # flash-attn is built from source against torch 2.8 (ABI lock).
#   # The two env vars below cap parallel ninja jobs and threads inside
#   # each nvcc call. Build is ~30 min on a 192-CPU node WITHOUT these,
#   # ~10-15 min WITH them.
#   # (TORCH_CUDA_ARCH_LIST is ignored by flash-attn setup.py, but listed
#   # for completeness.)
#   TORCH_CUDA_ARCH_LIST="8.9" MAX_JOBS=48 NVCC_THREADS=4 \
#       pip install -r requirements.txt
#

#     ("Failed to apply prompt replacement for mm_items['image'][0]") —
